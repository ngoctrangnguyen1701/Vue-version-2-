<div id="app">
    <p v-once>{{content}} - {{showContent()}}</p>
    <!-- v-once:  thực hiện các phép nội suy một lần 
    (không cập nhật lại khi dữ liệu thay đổi)

  </div>

  <script>
    new Vue({
      el: '#app',
      data: {
        content: 'First Content',
      },
      methods: {
        showContent: function(){
          this.content = 'Content has already changed'
          return this.content
        }
      }
    })
  </script>
